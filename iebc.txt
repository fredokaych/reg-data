        // ───────── Animated number ─────────
        function ANum({value,dur=700}){
            const[disp,setDisp]=useState(0);const prev=useRef(0);
            useEffect(()=>{const s=prev.current,e=value;if(s===e)return;const t0=performance.now();
            const run=now=>{const p=Math.min((now-t0)/dur,1);const ease=1-Math.pow(1-p,3);setDisp(Math.round(s+(e-s)*ease));
            if(p<1)requestAnimationFrame(run);else prev.current=e};requestAnimationFrame(run)},[value,dur]);
            return <span>{disp.toLocaleString()}</span>;
        }

        // ───────── Update panel ─────────
        function Update({data,setData,toast}){
            const[tab,setTab]=useState('quick');
            const[fDate,setFDate]=useState('');
            const[fVal,setFVal]=useState({kit1:'',kit2:'',kit3:''});
            const[editing,setEditing]=useState(false);
            const fRef=useRef(null);
            const[drag,setDrag]=useState(false);

            useEffect(()=>{const t=fmtD(new Date());const ok=DATES.some(d=>fmtD(d)===t);setFDate(ok?t:fmtD(DATES[0]))},[]);
            useEffect(()=>{const e=data[fDate];if(e){const has=e.kit1!==null||e.kit2!==null||e.kit3!==null;
            if(has){setFVal({kit1:e.kit1!==null?''+e.kit1:'',kit2:e.kit2!==null?''+e.kit2:'',kit3:e.kit3!==null?''+e.kit3:''});setEditing(true)}
            else{setFVal({kit1:'',kit2:'',kit3:''});setEditing(false)}}},[fDate,data]);

            const submit=()=>{
                const vs=KIT_KEYS.map(k=>fVal[k].trim());
                if(vs.every(v=>!v)){toast('Enter at least one kit value','error');return}
                const nums=vs.map(v=>v?parseInt(v):null);
                if(nums.some((n,i)=>vs[i]&&isNaN(n))){toast('Enter valid numbers','error');return}
                if(nums.some(n=>n!==null&&n<0)){toast('Values cannot be negative','error');return}
                if(!DATES.find(d=>fmtD(d)===fDate)){toast('Date outside registration period','error');return}
                const nd={...data};nd[fDate]={kit1:nums[0],kit2:nums[1],kit3:nums[2]};setData(nd);save(nd);
                toast((editing?'Updated':'Added')+' data for '+fmtS(DATES.find(d=>fmtD(d)===fDate)),'success');
                setFVal({kit1:'',kit2:'',kit3:''});setEditing(false);
                const ci=DATES.findIndex(d=>fmtD(d)===fDate);if(ci<DATES.length-1)setFDate(fmtD(DATES[ci+1]));
            };

            const parseCSV=file=>{if(!file)return;Papa.parse(file,{header:true,skipEmptyLines:true,complete:r=>{
                const nd={...data};let imp=0,skip=0;
                r.data.forEach(row=>{const dt=row['Date']||row['date'];if(!dt){skip++;return}
                if(!DATES.find(d=>fmtD(d)===dt)){skip++;return}
                const v1=row['Kit 1']||row['kit1']||row['kit_1']||'';
                const v2=row['Kit 2']||row['kit2']||row['kit_2']||'';
                const v3=row['Kit 3']||row['kit3']||row['kit_3']||'';
                nd[dt]={kit1:v1!==''?parseInt(v1):null,kit2:v2!==''?parseInt(v2):null,kit3:v3!==''?parseInt(v3):null};imp++});
                setData(nd);save(nd);toast('Imported '+imp+' entries'+(skip?', skipped '+skip:''),'success');
                if(fRef.current)fRef.current.value=''},error:err=>toast('CSV parse error: '+err.message,'error')})};

            const onDrop=e=>{e.preventDefault();setDrag(false);const f=e.dataTransfer.files[0];
            if(f&&f.name.endsWith('.csv'))parseCSV(f);else toast('Please upload a .csv file','error')};
            const dlSample=()=>{const b=new Blob([sampleCSV()],{type:'text/csv'});const a=document.createElement('a');a.href=URL.createObjectURL(b);a.download='wiga_sample_template.csv';a.click();toast('Sample template downloaded','info')};
            const dlExport=()=>{const b=new Blob([exportCSV(data)],{type:'text/csv'});const a=document.createElement('a');a.href=URL.createObjectURL(b);a.download='wiga_data_'+fmtD(new Date())+'.csv';a.click();toast('Data exported','success')};

            const[confirmReset,setConfirmReset]=useState(false);
            const doReset=()=>{const f=empty();setData(f);save(f);setConfirmReset(false);toast('All data cleared','success')};

                    {[['quick','fa-pen-to-square','Quick Entry'],['csv','fa-file-csv','CSV Upload'],['more','fa-ellipsis','More']].map(([k,ic,lb])=>
                    <button key={k} className={`tab-btn ${tab===k?'tab-active':'tab-inactive'}`} onClick={()=>setTab(k)}><i className={`fas ${ic}`} style={{marginRight:6}}/>{lb}</button>)}
                </div>
                <div style={{padding:18}}>
                    {tab==='quick'&&(<div className="fade-in">
                        {editing&&<div style={{marginBottom:14,padding:'9px 14px',background:'rgba(245,158,11,.07)',borderRadius:9,border:'1px solid rgba(245,158,11,.12)'}}>
                            <span style={{fontSize:12,color:'#f59e0b'}}><i className="fas fa-pen" style={{marginRight:6}}/>Editing existing entry — submit to overwrite</span></div>}
                        <div style={{display:'grid',gridTemplateColumns:'repeat(auto-fit,minmax(160px,1fr))',gap:10,alignItems:'end'}}>

        // ───────── Doughnut chart ─────────
        function DonutChart({data}){
            const cv=useRef(null);const ch=useRef(null);
            useEffect(()=>{if(ch.current)ch.current.destroy();
                const totals=KIT_KEYS.map(k=>DATES.reduce((s,d)=>{const e=data[fmtD(d)];return s+(e&&e[k]!==null?e[k]:0)},0));
                const has=totals.some(t=>t>0);
                ch.current=new Chart(cv.current.getContext('2d'),{type:'doughnut',data:{labels:KIT_NAMES,datasets:[{data:has?totals:[.001,.001,.001],
                backgroundColor:['rgba(16,185,129,.78)','rgba(245,158,11,.78)','rgba(244,63,94,.78)'],borderColor:'#0f1612',borderWidth:3,hoverOffset:6}]},
                options:{responsive:true,maintainAspectRatio:false,cutout:'66%',
                    plugins:{legend:{position:'bottom',labels:{color:'rgba(226,232,240,.55)',font:{family:'DM Sans',size:11},boxWidth:10,boxHeight:10,borderRadius:2,useBorderRadius:true,padding:14}},
                    tooltip:{backgroundColor:'#151e19',borderColor:'rgba(16,185,129,.15)',borderWidth:1,titleColor:'#e2e8f0',bodyColor:'rgba(226,232,240,.65)',titleFont:{family:'DM Sans',weight:'600'},bodyFont:{family:'DM Sans'},padding:10,cornerRadius:8,
                    callbacks:{label:ctx=>{if(!has)return'No data';const tot=totals.reduce((a,b)=>a+b,0);return ctx.label+': '+ctx.raw.toLocaleString()+' ('+((ctx.raw/tot)*100).toFixed(1)+'%)'}}}}}});
                return()=>{if(ch.current)ch.current.destroy()}},[data]);
            return(<div className="glass-card fade-in" style={{padding:18,animationDelay:'560ms'}}>
                <h3 className="font-display" style={{fontSize:15,fontWeight:700,color:'#e2e8f0',marginBottom:14}}>Kit Performance</h3>
                <div style={{height:310,position:'relative',display:'flex',alignItems:'center',justifyContent:'center'}}><canvas ref={cv}/></div>
            </div>);
        }

        // ───────── Data table ─────────
        function Table({data,setData,toast}){
            const[editK,setEditK]=useState(null);const[ev,setEv]=useState({kit1:'',kit2:'',kit3:''});
            let cum=0;
            const rows=DATES.map((d,i)=>{const k=fmtD(d);const e=data[k]||{};const dt=[e.kit1,e.kit2,e.kit3].reduce((a,b)=>(a||0)+(b||0),0);cum+=dt;
            return{i:i+1,d,k,e,dt,cum,has:dt>0,today:k===fmtD(new Date())}});

            const startEdit=r=>{setEditK(r.k);setEv({kit1:r.e.kit1!==null?''+r.e.kit1:'',kit2:r.e.kit2!==null?''+r.e.kit2:'',kit3:r.e.kit3!==null?''+r.e.kit3:''})};
            const saveRow=()=>{const nd={...data};nd[editK]={kit1:ev.kit1!==''?parseInt(ev.kit1):null,kit2:ev.kit2!==''?parseInt(ev.kit2):null,kit3:ev.kit3!==''?parseInt(ev.kit3):null};
            setData(nd);save(nd);setEditK(null);toast('Entry updated','success')};

            return(<div className="glass-card fade-in" style={{padding:0,animationDelay:'640ms'}}>
                <div style={{padding:'18px 18px 14px',borderBottom:'1px solid rgba(16,185,129,.05)'}}>
                    <h3 className="font-display" style={{fontSize:15,fontWeight:700,color:'#e2e8f0'}}>Detailed Records</h3>
                    <p style={{fontSize:11,color:'rgba(226,232,240,.3)',marginTop:2}}>Click any row to edit inline</p>
                </div>
                <div className="table-wrap" style={{maxHeight:500,overflowY:'auto'}}>
                    <table className="data-table" style={{width:'100%',borderCollapse:'collapse',fontSize:13}}>
                        <thead><tr>
                            {['#','Date','Day',...KIT_NAMES,'Daily','Cumulative'].map((h,hi)=>(
                            <th key={hi} style={{padding:'11px 14px',textAlign:hi>=3?'right':'left',fontSize:10,fontWeight:600,textTransform:'uppercase',letterSpacing:'.06em',color:hi>=3&&hi<6?KIT_CLR[hi-3]:'rgba(226,232,240,.35)',borderBottom:'1px solid rgba(16,185,129,.05)'}}>{h}</th>
                            ))}
                        </tr></thead>
                        <tbody>{rows.map((r,idx)=>{
                            const isEditing=editK===r.k;
                            return(<tr key={r.k} onClick={()=>!isEditing&&startEdit(r)} style={{cursor:'pointer',background:r.today?'rgba(16,185,129,.035)':idx%2?'rgba(16,185,129,.012)':'transparent',borderBottom:'1px solid rgba(16,185,129,.03)',transition:'background .12s'}}
                                onMouseEnter={e=>{if(!isEditing)e.currentTarget.style.background='rgba(16,185,129,.055)'}}
                                onMouseLeave={e=>{e.currentTarget.style.background=r.today?'rgba(16,185,129,.035)':idx%2?'rgba(16,185,129,.012)':'transparent'}}>
                                <td style={{padding:'9px 14px',color:'rgba(226,232,240,.25)',fontSize:11}}>{r.i}</td>
                                <td style={{padding:'9px 14px',color:r.today?'#10b981':'rgba(226,232,240,.65)',fontWeight:r.today?600:400,whiteSpace:'nowrap'}}>
                                    {fmtS(r.d)}{r.today&&<span style={{fontSize:9,color:'#10b981',marginLeft:5,fontWeight:500,border:'1px solid rgba(16,185,129,.2)',borderRadius:4,padding:'1px 5px'}}>TODAY</span>}
                                </td>
                                <td style={{padding:'9px 14px',color:'rgba(226,232,240,.3)',fontSize:11}}>{DAYS[r.d.getDay()]}</td>
                                {isEditing?KIT_KEYS.map(k=>(<td key={k} style={{padding:'5px 6px'}}>
                                    <input type="number" className="input-field" style={{padding:'5px 7px',fontSize:12,textAlign:'right',width:66}} value={ev[k]} min="0"
                                    onChange={e=>setEv({...ev,[k]:e.target.value})} onClick={e=>e.stopPropagation()} onKeyDown={e=>{if(e.key==='Enter')saveRow();if(e.key==='Escape')setEditK(null)}} autoFocus/>
                                </td>)):KIT_KEYS.map((k,ki)=>(<td key={k} style={{padding:'9px 14px',textAlign:'right',color:r.e[k]!==null?KIT_CLR[ki]:'rgba(226,232,240,.12)',fontWeight:r.e[k]!==null?600:400}}>
                                    {r.e[k]!==null?r.e[k].toLocaleString():'—'}
                                </td>))}
                                {isEditing?(
                                    <>
                                    <td style={{padding:'9px 14px',textAlign:'right'}}><button className="btn-primary" style={{padding:'4px 12px',fontSize:11,marginRight:4}} onClick={e=>{e.stopPropagation();saveRow()}}>Save</button><button className="btn-secondary" style={{padding:'4px 12px',fontSize:11}} onClick={e=>{e.stopPropagation();setEditK(null)}}>Cancel</button></td>
                                    <td></td>
                                    </>
                                ):(
                                    <>
                                    <td style={{padding:'9px 14px',textAlign:'right',color:r.dt>0?'#e2e8f0':'rgba(226,232,240,.12)',fontWeight:r.dt>0?700:400}}>{r.dt>0?r.dt.toLocaleString():'—'}</td>
                                    <td style={{padding:'9px 14px',textAlign:'right',color:r.cum>0?'rgba(226,232,240,.5)':'rgba(226,232,240,.12)',fontWeight:500}}>{r.cum>0?r.cum.toLocaleString():'—'}</td>
                                    </>
                                )}
                            </tr>)})}</tbody>
                    </table>
                </div>
            </div>);
        }
